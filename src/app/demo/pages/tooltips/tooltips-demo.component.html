{{ 'TOOLTIP.API.EXAMPLE_TITLE' | translate }}

import { PshTooltipComponent } from 'ps-helix';

@Component({
  imports: [PshTooltipComponent],
  template: `
    <psh-tooltip
      content="Additional information"
      variant="dark"
      position="top"
      [showDelay]="200"
      [hideDelay]="100"
      [maxWidth]="250"
      [autoFlip]="true"
      (shown)="onTooltipShow()"
      (hidden)="onTooltipHide()"
    >
      <psh-button>Hover for info</psh-button>
    </psh-tooltip>
  `
})
export class MyComponent {
  onTooltipShow(): void {
    console.log('Tooltip shown');
  }

  onTooltipHide(): void {
    console.log('Tooltip hidden');
  }
}
